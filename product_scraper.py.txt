import requests
from bs4 import BeautifulSoup

def get_product_links(category_url, max_pages=3):
    product_links = []
    for page in range(1, max_pages + 1):
        url = f"{category_url}?page={page}"
        res = requests.get(url)
        if res.status_code != 200:
            break
        soup = BeautifulSoup(res.text, 'html.parser')
        items = soup.select('div.product-card a')
        for a in items:
            href = a.get('href')
            if href:
                product_links.append('https://wiraa.ir' + href)
    return product_links

if __name__ == "__main__":
    category = "https://wiraa.ir/category/آرایشی-و-بهداشتی"
    links = get_product_links(category)
    for link in links:
        print(link)
